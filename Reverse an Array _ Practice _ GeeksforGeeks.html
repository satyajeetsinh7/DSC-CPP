class Solution {
  public:
  
    void solve(vector<int> &arr, int left, int right) {
        if (left >= right) return;   

        swap(arr[left], arr[right]); 
        solve(arr, left + 1, right - 1); 
    }

    void reverseArray(vector<int> &arr) {
        solve(arr, 0, arr.size() - 1);
    }
};
